##### 1. 项目卡片点击跳转功能

**文件：** `index.html`

**实现内容：**
- 为所有 `.project-card` 元素添加点击事件监听器
- 点击卡片任意位置（除链接外）均可跳转到对应的项目详情页
- 使用 `window.open()` 在新标签页打开详情页
- 自动从 `data-project-id` 属性提取项目 ID，转换为详情页 URL
- 转换规则：`project1` → `work1.html`，`project2` → `work2.html`，以此类推

**关键代码：**
```javascript
// 项目卡片点击跳转
document.querySelectorAll('.project-card').forEach(card => {
    card.addEventListener('click', function(e) {
        // 如果点击的是链接本身，不阻止默认行为
        if (e.target.tagName === 'A' || e.target.closest('a')) {
            return;
        }
        
        e.preventDefault();
        const projectId = this.getAttribute('data-project-id');
        const projectName = this.getAttribute('data-project-name');
        
        // 追踪点击事件
        if (window.Analytics) {
            window.Analytics.trackProjectInteraction(projectName, 'card_click', {
                location: 'project_list',
                target: `works/${projectId.replace('project', 'work')}.html`
            });
        }
        
        // 跳转到详情页
        if (projectId) {
            const targetUrl = `works/${projectId.replace('project', 'work')}.html`;
            window.open(targetUrl, '_blank');
        }
    });
    
    // 添加鼠标指针样式
    card.style.cursor = 'pointer';
});
```

**数据追踪：**
- 使用 `trackProjectInteraction()` 记录项目交互事件
- 记录项目名称、交互类型（card_click）、位置（project_list）
- 记录目标 URL 信息

##### 2. AI Demo 卡片点击跳转功能

**文件：** `index.html`

**实现内容：**
- 为所有 `.demo-card` 元素添加点击事件监听器
- 点击卡片任意位置（除链接外）均可跳转到对应的 Demo 链接
- 使用 `window.open()` 在新标签页打开外部 Demo 链接
- 自动从卡片中提取 `data-project-id` 和 `data-project-name` 属性
- 查找卡片内的 `<a>` 标签获取目标 URL

**关键代码：**
```javascript
// AI Demo 卡片点击跳转
document.querySelectorAll('.demo-card').forEach(card => {
    card.addEventListener('click', function(e) {
        // 如果点击的是链接本身，不阻止默认行为
        if (e.target.tagName === 'A' || e.target.closest('a')) {
            return;
        }
        
        e.preventDefault();
        const demoId = this.getAttribute('data-project-id');
        const demoName = this.getAttribute('data-project-name');
        const linkElement = this.querySelector('a[href]');
        
        // 追踪点击事件
        if (window.Analytics) {
            window.Analytics.trackClick(
                `ai-demo-${demoId}`,
                demoName || 'AI Demo',
                'demo_card',
                {
                    location: 'demo_section',
                    target: linkElement?.href || '#'
                }
            );
        }
        
        // 跳转到 Demo 链接
        if (linkElement && linkElement.href) {
            window.open(linkElement.href, '_blank');
        }
    });
    
    // 添加鼠标指针样式
    card.style.cursor = 'pointer';
});
```

**数据追踪：**
- 使用 `trackClick()` 记录 Demo 卡片点击事件
- 记录 Demo ID、Demo 名称、元素类型（demo_card）
- 记录位置（demo_section）和目标 URL

##### 3. 修复交互趋势图表无数据问题

**文件：** `works/analytics-dashboard.html`

**问题分析：**
- `renderInteractionTrend()` 函数原本直接从 localStorage 获取数据
- 但在 `renderAnalytics()` 函数中调用时，没有传递数据参数
- 导致函数内部获取数据的方式不一致，可能出现数据为空的情况

**解决方案：**
- 修改 `renderInteractionTrend()` 函数签名，接收 data 参数
- 在 `renderAnalytics()` 中调用时传递 data 参数
- 确保数据来源一致，都来自 `calculateStats()` 函数的输入数据

**修改代码：**
```javascript
// 修改前
function renderInteractionTrend() {
    const data = Analytics.getStoredData ? Analytics.getStoredData() : getStoredData();
    // ...
}

// 修改后
function renderInteractionTrend(data) {
    if (!data || data.length === 0) {
        container.innerHTML = '<p class="text-gray-500 text-center py-8">暂无趋势数据</p>';
        return;
    }
    // ...
}

// 调用处修改
renderInteractionTrend(data); // 传递 data 参数
```

// 在 calculateProjectStats 函数中
const sessionDemoClicks = {};

// Demo 链接点击（二元统计：每个会话只统计一次）
if (event.eventType === 'demo_link_click') {
    sessionDemoClicks[event.sessionId] = true;
}

// 计算 Demo 点击率：点击的会话数 / 总会话数
const uniqueSessions = new Set(data.map(e => e.sessionId)).size;
const clickedSessions = Object.keys(sessionDemoClicks).length;
const demoClickRate = uniqueSessions > 0 ? Math.round((clickedSessions / uniqueSessions) * 100) : 0;

// 返回统计数据
return {
    pageViews,
    totalInteractions,
    sectionViews,
    documentViews,
    timeSpentData,
    demoClickRate  // 新增字段
};

7. **文件下载 (File Download)** - 追踪简历下载情况
